cmake_minimum_required(VERSION 2.6)
project(UAS_OS)

SET( ARMADILLO_INCLUDE_DIRS  /usr/include/armadillo-bits/)
SET( ncurses_INCLUDE_DIRS /usr/include/ncurses5-config)
SET( ncurses_LIBS /lib/libncurses.so.5 /lib/libncurses.so.5.7 /lib/libncursesw.so.5 /lib/libncursesw.so.5.7)
# find required packages
find_package(OpenCV REQUIRED)
#find_package(ncurses5-dev REQUIRED)
#find_package(armadillo_bits REQUIRED)

# set build directory
set(CMAKE_CURRENT_BINARY_DIR ${PROJECT_SOURCE_DIR}/build)

# general settings for all sub projects
set(EXECUTABLE_OUTPUT_PATH ${PROJECT_SOURCE_DIR}/bin)
set(LIBRARY_OUTPUT_PATH ${PROJECT_SOURCE_DIR}/lib)

#set the include directories
include_directories (${PROJECT_SOURCE_DIR}/include  ${PROJECT_SOURCE_DIR}/include/UAS_tld ${PROJECT_SOURCE_DIR}/include/UAS_nav  ${OpenCV_INCLUDE_DIRS} ${ncurses_INCLUDE_DIRS} ${ARMADILLO_INCLUDE_DIRS})

# add sub projects
add_subdirectory(UAS_tld)
add_subdirectory(UAS_nav)
add_subdirectory(UAS_main)
